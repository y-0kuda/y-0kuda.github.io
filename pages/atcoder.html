Prime Number

Check all numbers up to √n to determine whether n is divisible.

bool isPrime(long long n){
    if(n<2)return false;
    // If n has any divisor, at least one of them is always less than or equal to √n.
    for(long long i=2;i*i<=n;i++)
        if(n%i==0)return false;
    return true;
}

Sieve of Eratosthenes

Find all prime numbers from 1 to N. Eliminate all numbers that are not prime.

// Assume all numbers from 0 to N are prime (true) initially.
vector<bool> isp(N+1,true);
// Remove 0 and 1, since they are not prime from the start.
isp[0]=isp[1]=false;
for(int i=2;i*i<=N;i++)
 if(isp[i])
  // Check multiples (e.g., (2), 4, 6 and (3), 6, 9).
  for(int j=i*i;j<=N;j+=i) isp[j]=false;

vector<pair<long long,int>>,greater<>> pq;

dist[s]=0;pq.push({0,s});
while(!pq.empty()){
 auto [d,v]=pq.top();pq.pop();
 if(d>dist[v])continue;
